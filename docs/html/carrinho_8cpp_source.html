158void Carrinho::processaSerial()
159{
160 static size_t n = 0;
161
162 auto skipWS = [](const char *s) -> const char *
163 {
164 while (*s && isspace((unsigned char)*s))
165 ++s;
166 return s;
167 };
168
169 auto parseFloatAfter = [&](const char *s, size_t prefixLen, float &out) -> bool
170 {
171 const char *p = skipWS(s + prefixLen);
172 char *end = nullptr;
173 float v = strtof(p, &end);
174 if (end == p)
175 return false; // nada lido
176 out = v;
177 return true;
178 };
179
180 auto parseLongAfter = [&](const char *s, size_t prefixLen, long &out) -> bool
181 {
182 const char *p = skipWS(s + prefixLen);
183 char *end = nullptr;
184 long v = strtol(p, &end, 10);
185 if (end == p)
186 return false; // nada lido
187 out = v;
188 return true;
189 };
190
191 while (Serial.available())
192 {
193 char c = (char)Serial.read();
194
195 // fim de linha delimita o comando
196 if (c == '\n' || c == '\r')
197 {
198 cmdBuf[n] = '\0';
199 n = 0;
200 if (cmdBuf[0] == '\0')
201 return;
202
203 if (strcmp(cmdBuf, "h") == 0)
204 {
205 printHelp();
206 }
207 else if (strcmp(cmdBuf, "p") == 0)
208 {
209 Serial.printf(
210 "vy=%.1f kp=%.3f ki=%.3f kd=%.3f om=%.1f log=%s/%lums invOmega=%s\n",
211 vyPercent, kp, ki, kd, omegaMax,
212 LOG_ATIVO ? "on" : "off",
213 (unsigned long)LOG_MS,
214 INVERTER_OMEGA ? "true" : "false");
215 }
216 else if (strcmp(cmdBuf, "r") == 0)
217 {
218 iniciarSeguirLinha();
219 Serial.println("estado: ESPERANDO_LARGADA");
220 }
221 else if (strcmp(cmdBuf, "s") == 0)
222 {
223 entrarCalibracao();
224 Serial.println("estado: CALIBRACAO");
225 }
226 else if (cmdBuf[0] == 'v' && cmdBuf[1] == 'y')
227 {
228 float v;
229 if (parseFloatAfter(cmdBuf, 2, v))
230 {
231 v = constrain(v, 0.0f, 100.0f);
232 vyPercent = v;
233 Serial.printf("vy = %.1f\n", vyPercent);
234 }
235 else
236 Serial.println("vy invalido");
237 }
238 else if (cmdBuf[0] == 'k' && cmdBuf[1] == 'p')
239 {
240 float v;
241 if (parseFloatAfter(cmdBuf, 2, v) && v > 0.0f)
242 {
243 kp = v;
244 Serial.printf("kp = %.3f\n", kp);
245 }
246 else
247 Serial.println("kp invalido");
248 }
249 else if (cmdBuf[0] == 'k' && cmdBuf[1] == 'i')
250 {
251 float v;
252 if (parseFloatAfter(cmdBuf, 2, v) && v >= 0.0f)
253 {
254 ki = v;
255 Serial.printf("ki = %.3f\n", ki);
256 }
257 else
258 Serial.println("ki invalido");
259 }
260 else if (cmdBuf[0] == 'k' && cmdBuf[1] == 'd')
261 {
262 float v;
263 if (parseFloatAfter(cmdBuf, 2, v) && v >= 0.0f)
264 {
265 kd = v;
266 Serial.printf("kd = %.3f\n", kd);
267 }
268 else
269 Serial.println("kd invalido");
270 }
271 else if (cmdBuf[0] == 'l' && cmdBuf[1] == 'o' && cmdBuf[2] == 'g')
272 {
273 if (strcmp(cmdBuf, "log0") == 0)
274 {
275 LOG_ATIVO = false;
276 Serial.println("log off");
277 }
278 else
279 {
280 long ms;
281 if (parseLongAfter(cmdBuf, 3, ms) && ms >= 10)
282 {
283 LOG_MS = (uint32_t)ms;
284 LOG_ATIVO = true;
285 Serial.printf("log every %lums\n", (unsigned long)LOG_MS);
286 }
287 else
288 {
289 LOG_ATIVO = true;
290 Serial.println("log on");
291 }
292 }
293 }
294 else if (strcmp(cmdBuf, "inv") == 0)
295 {
296 INVERTER_OMEGA = !INVERTER_OMEGA;
297 Serial.printf("INVERTER_OMEGA = %s\n", INVERTER_OMEGA ? "true" : "false");
298 }
299 else if (cmdBuf[0] == 'o' && cmdBuf[1] == 'm')
300 {
301 float v;
302 if (parseFloatAfter(cmdBuf, 2, v) && v > 0.0f)
303 {
304 omegaMax = v;
305 Serial.printf("omegaMax = %.1f\n", omegaMax);
306 }
307 else
308 Serial.println("omegaMax invalido");
309 }
310 else
311 {
312 Serial.println("comando nao reconhecido. use h para ajuda.");
313 }
314 }
315 else if (n < sizeof(cmdBuf) - 1)
316 {
317 cmdBuf[n++] = c; // acumula caractere no buffer do comando
318 }
319 }
320}

330void Carrinho::seguidorInitLUT()
331{
332 static const int8_t PESOS[8] = {-7, -5, -3, -1, 1, 3, 5, 7};
333 for (int m = 0; m < 256; m++)
334 {
335 uint8_t active = PRETO_BIT_1 ? (uint8_t)m : (uint8_t)(~m);
336 if (active == 0)
337 {
338 erroLUT[m] = (int8_t)0x7F; // marca "sem linha"
339 continue;
340 }
341 int soma = 0, cnt = 0;
342 for (int i = 0; i < 8; i++)
343 if (active & (1u << i))
344 {
345 soma += PESOS[i];
346 cnt++;
347 }
348
349 // média arredondada para o inteiro mais próximo
350 int e = (soma >= 0) ? (soma + cnt / 2) / cnt : (soma - cnt / 2) / cnt;
351 erroLUT[m] = (int8_t)e;
352 }
353}

465void Carrinho::acionaMotor(int i, int velPct)
466{
467 if (i < 0 || i > 3)
468 return;
469 if (velPct > 100)
470 velPct = 100;
471 else if (velPct < -100)
472 velPct = -100;
473
474 if (invertMotor[i])
475 velPct = -velPct;
476
477 uint8_t duty = porcentagemPWM((uint8_t)abs(velPct));
478 if (duty == 0)
479 {
480 writePWM(i, 0, 0);
481 writePWM(i, 1, 0);
482 return;
483 }
484
485 if (velPct >= 0)
486 {
487 writePWM(i, 0, duty);
488 writePWM(i, 1, 0);
489 }
490 else
491 {
492 writePWM(i, 0, 0);
493 writePWM(i, 1, duty);
494 }
495}

504void Carrinho::acionaRodasOminiInt(int vx, int vy, int omega)
505{
506 int v0 = vy + vx + omega;
507 int v1 = vy - vx - omega;
508 int v2 = vy - vx + omega;
509 int v3 = vy + vx - omega;
510
511 int a0 = abs(v0), a1 = abs(v1), a2 = abs(v2), a3 = abs(v3);
512 int m = max(max(a0, a1), max(a2, a3));
513 if (m > 10000)
514 {
515 v0 = (int)((int64_t)v0 * 10000 / m);
516 v1 = (int)((int64_t)v1 * 10000 / m);
517 v2 = (int)((int64_t)v2 * 10000 / m);
518 v3 = (int)((int64_t)v3 * 10000 / m);
519 }
520
521 auto iround = [](int x) -> int
522 { return (x >= 0 ? (x + 50) / 100 : (x - 50) / 100); };
523 acionaMotor(0, iround(v0));
524 acionaMotor(1, iround(v1));
525 acionaMotor(2, iround(v2));
526 acionaMotor(3, iround(v3));
527}

566float Carrinho::pidAtualizar(float erro, float dt)
567{
568 // I com anti windup simples
569 integralAcumulada += erro * dt;
570 float lim = 100.0f / (ki > 0.0f ? ki : 0.001f);
571 integralAcumulada = constrain(integralAcumulada, -lim, lim);
572
573 // D com filtro RC
574 float derivadaBruta = (erro - erroAnterior) / (dt > 0.0f ? dt : 0.001f);
575 float alpha = dt / (tauD + dt);
576 dFilt = dFilt + alpha * (derivadaBruta - dFilt);
577 erroAnterior = erro;
578
579 float omega = kp * erro + ki * integralAcumulada + kd * dFilt; // aqui estava "derivada"
580 omega = constrain(omega, -omegaMax, omegaMax);
581 return omega;
582}

630void Carrinho::modoCorrida(uint8_t, float erro, float dt)
631{
632 if (erro == ERRO_SEM_LINHA)
633 {
634 int vy_c = toCent(10.0f);
635 int om_c = (ultimoErroValido >= 0.0f)
636 ? toCent(INVERTER_OMEGA ? -12.0f : +12.0f)
637 : toCent(INVERTER_OMEGA ? +12.0f : -12.0f);
638 acionaRodasOminiInt(0, vy_c, om_c); // movimento de varredura
639 logStatus(ERRO_SEM_LINHA, 0.0f);
640 return;
641 }
642
643 ultimoErroValido = erro;
644 float omega = pidAtualizar(erro, dt);
645 if (INVERTER_OMEGA)
646 omega = -omega;
647
648 controlarRodas(vyPercent, 0.0f, omega);
649 logStatus(erro, omega);
650}

679void Carrinho::seguirLinhaStep()
680{
681 static uint32_t lastMicros = micros();
682 uint32_t now = micros();
683 float dt = clampDt((now - lastMicros) * 1e-6f);
684 lastMicros = now;
685
686 float erro = lerErro();
687 if (erro == ERRO_SEM_LINHA)
688 {
689 controlarRodas(10.0f, 0.0f,
690 (ultimoErroValido >= 0.0f)

693 logStatus(ERRO_SEM_LINHA, 0.0f);
694 return;
695 }
696
697 ultimoErroValido = erro;
698 float omega = pidAtualizar(erro, dt);
699 if (INVERTER_OMEGA)
700 omega = -omega;
701
702 controlarRodas(vyPercent, 0.0f, omega);
703 logStatus(erro, omega);

724void Carrinho::tick()
725{
726 processaSerial();
727
728 uint32_t t = micros();
729 float dt = clampDt((t - tPrevMicros) * 1e-6f);
730 tPrevMicros = t;
731
732 uint8_t mascara = lerLinhaMascara();
733 float erro = calcularErroMascara(mascara);
734
735 switch (estado)
736 {
737 case CALIBRACAO:
738 modoCalibracao(mascara, erro);
739 break;
740 case ESPERANDO_LARGADA:
741 modoEsperandoLargada(mascara);
742 break;
743 case CORRIDA:
744 modoCorrida(mascara, erro, dt);
745 break;
746 }
747}
